```java
class Solution {
    public int findKthNumber(int n, int k) {
        long current = 1;
        k--; // Convert to 0-based index
        
        while (k > 0) {
            int steps = countSteps(current, current + 1, n);
            if (steps <= k) {
                current++;
                k -= steps;
            } else {
                current *= 10;
                k--;
            }
        }
        return (int)current;
    }
    
    private int countSteps(long n1, long n2, int max) {
        int steps = 0;
        while (n1 <= max) {
            steps += Math.min(n2, max + 1) - n1;
            n1 *= 10;
            n2 *= 10;
        }
        return steps;
    }
}
```

```java
class Solution {
```
