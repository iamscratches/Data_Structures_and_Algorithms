class Solution {
    public boolean isPowerOfThree(int n) {
        // Handle edge cases (non-positive numbers)
        if (n <= 0) {
            return false;
        }
        
        // Continuously divide by 3 while divisible
        while (n != 1) {
            if (n % 3 != 0) {
                return false;
            }
            n /= 3;
        }
        
        return true;
    }
}
