class Solution {
    public String stringFilter(String str) {
        StringBuilder sb = new StringBuilder();
        int i = 0;
        while (i < str.length()) {
            if (i+1 < str.length() && str.charAt(i) == 'a' && str.charAt(i+1) == 'c') {
                i += 2;
            } else if (str.charAt(i) == 'b') {
                i++;
            } else {
                sb.append(str.charAt(i++));
            }
        }
        return sb.toString();
    }
